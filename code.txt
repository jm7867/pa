    <script>
        /* Globals */
        let userCoords = null;
        let todayPrayerTimes = []; // { key, name, time (Date), timeStr24 }
        let azanPlayedFor = null; // key of prayer already played
        let isMuted = false;
        let selectedDateGlobal = new Date();
        const audio = document.getElementById('azan-audio');
        audio.muted = isMuted;
        // Unlock audio on first user click
        document.addEventListener('click', async () => {
            try {
                await audio.play();
                audio.pause();
                audio.currentTime = 0;
            } catch (e) {
                console.warn('Audio unlock failed', e);
            }
        });
        /* DOM helpers */
        const $ = id => document.getElementById(id);
        /* Digital clock */
        function updateDigitalClock() {
            const now = new Date();
            let hours = now.getHours();
            const minutes = now.getMinutes();
            const seconds = now.getSeconds();
            const ampm = hours >= 12 ? 'PM' : 'AM';
            hours = hours % 12 || 12;
            const timeStr = `${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
            $('digital-clock').innerHTML = `${timeStr}<br><span style="font-size:40px">${ampm}</span>`;
        }
        setInterval(updateDigitalClock, 1000);
        updateDigitalClock();
        /* Toggle mute/unmute: attempt to unlock audio on unmute (user interaction) */
        $('mute-btn').addEventListener('click', async () => {
            isMuted = !isMuted;
            audio.muted = isMuted;
            $('mute-btn').textContent = isMuted ? '🔇' : '🔊';
            if (!isMuted) {
                try {
                    await audio.play();
                    audio.pause();
                    audio.currentTime = 0;
                } catch (e) {
                    /* won't break if autoplay blocked */
                }
            }
        });
        /* Refresh button */
        $('refresh-btn').addEventListener('click', () => {
            if (userCoords) {
                fetchPrayerTimes(selectedDateGlobal);
                fetchWeather(userCoords.latitude, userCoords.longitude);
                location.reload();
            }
        });
        /* Format time to 12h (HH:MM AM/PM), accepts "HH:MM" or "HH:MM:SS" */
        function formatTime12h(timeStr) {
            const parts = timeStr.split(':').map(Number);
            let h = parts[0],
                m = parts[1] || 0;  // ← Single || (logical OR)
            const ampm = h >= 12 ? 'PM' : 'AM';
            h = h % 12 || 12;
            return `${h}:${String(m).padStart(2, '0')} ${ampm}`;
        }

        /* Utility to parse time string "HH:MM[:SS]" into Date on given date */
        function getTimeObj(timeStr, date = selectedDateGlobal) {
            const [h, m, s = '0'] = timeStr.split(':').map(Number);
            const t = new Date(date);
            t.setHours(Number(h), Number(m), Number(s || 0), 0);
            return t;
        }
        async function fetchWeather(lat, lon) {
            try {
                const url = `https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current=temperature_2m,wind_speed_10m&temperature_unit=fahrenheit`;
                const res = await fetch(url);
                const data = await res.json();
                $('weather-temp').textContent = `🌡 ${data.current.temperature_2m}°F`;
                $('weather-wind').textContent = `💨 ${data.current.wind_speed_10m} km/h`;
            } catch {
                $('weather-temp').textContent = '--';
                $('weather-wind').textContent = '--';
            }
        }
        /* Main: fetch prayer times from Aladhan & render */
        async function fetchPrayerTimes(selectedDate = new Date()) {
            if (!navigator.onLine) {
                console.warn('Offline – skipping prayer time fetch');
                $('current-prayer').textContent = 'Offline';
                return;
            }
            if (!userCoords) return;
            if (!userCoords) return;
            selectedDateGlobal = selectedDate;
            // Reset todayPrayerTimes
            todayPrayerTimes = [];
            azanPlayedFor = null; // reset for new day/load
            const lat = userCoords.latitude;
            const lon = userCoords.longitude;
            const method = 2; // you can expose this
            const day = selectedDate.getDate();
            const month = selectedDate.getMonth() + 1;
            const year = selectedDate.getFullYear();
            const url = `https://api.aladhan.com/v1/calendar?latitude=${lat}&longitude=${lon}&method=${method}&month=${month}&year=${year}&school=1`;
            try {
                const res = await fetch(url);
                const json = await res.json();
                if (!json || json.code !== 200 || !json.data) throw new Error('Aladhan API error');
                const today = json.data[day - 1];
                const timings = today.timings;
                const hijri = today.date.hijri;
                // Format Gregorian date nicely
                const dateObj = new Date(selectedDate);
                const options = {
                    weekday: 'short',
                    year: 'numeric',
                    month: 'long',
                    day: 'numeric'
                };
                const formattedDate = dateObj.toLocaleDateString('en-US', {
                    weekday: 'short',
                    month: 'short',
                    day: '2-digit',
                    year: 'numeric'
                }).replace(',', '');
                // Example output: Saturday, August 16, 2025
                $('gregorian-date').textContent = formattedDate;
                // Hijri (from API)
                $('hijri-date').textContent = `${hijri.day} ${hijri.month.en} ${hijri.year}`;
                $('hijri-date').textContent = `${hijri.day} ${hijri.month.en} ${hijri.year}`;
                // prayer order (Ishraq derived)
                const prayerOrder = [{
                    name: "Imsak 🌙",
                    key: "Imsak"
                }, {
                    name: "Fajr ✨",
                    key: "Fajr"
                }, {
                    name: "Sunrise 🌅",
                    key: "Sunrise"
                }, {
                    name: "Ishraq 🌄",
                    key: "Ishraq",
                    derived: true,
                    afterMinutes: 22
                }, {
                    name: "Dhuhr ☀️",
                    key: "Dhuhr"
                }, {
                    name: "Asr ☁️",
                    key: "Asr"
                }, {
                    name: "Maghrib 🌇",
                    key: "Maghrib"
                }, {
                    name: "Isha’a 🌙",
                    key: "Isha"
                }];
                const prayerList = $('prayer-list');
                prayerList.innerHTML = '';
                const now = new Date();
                const nowTime = new Date(now.getFullYear(), now.getMonth(), now.getDate(), now.getHours(), now.getMinutes(), now.getSeconds());
                todayPrayerTimes = []; // reset
                // Build prayer times array
                for (const prayer of prayerOrder) {
                    let timeStr24;
                    if (prayer.derived && prayer.key === 'Ishraq') {
                        const sunriseRaw = timings['Sunrise'].split(' ')[0]; // HH:MM
                        const [sh, sm] = sunriseRaw.split(':').map(Number);
                        const total = sh * 60 + sm + (prayer.afterMinutes || 15);
                        const ih = Math.floor(total / 60) % 24;
                        const im = total % 60;
                        timeStr24 = `${String(ih).padStart(2, '0')}:${String(im).padStart(2, '0')}:00`;
                    } else {
                        if (!timings[prayer.key]) continue;
                        timeStr24 = timings[prayer.key].split(' ')[0];
                        if (timeStr24.length === 5) timeStr24 += ':00';
                    }
                    const prayerTime = getTimeObj(timeStr24, selectedDate);
                    todayPrayerTimes.push({
                        key: prayer.key,
                        name: prayer.name,
                        time: prayerTime,
                        timeStr24
                    });
                }
                // 🔹 Render list with NOW & FINISHED logic
                let currentPrayer = null;
                for (let i = 0; i < todayPrayerTimes.length; i++) {
                    const prayer = todayPrayerTimes[i];
                    const nextPrayer = todayPrayerTimes[i + 1]; // look ahead
                    let isNow = false;
                    let isFinished = false;
                    if (nowTime >= prayer.time) {
                        if (!nextPrayer || nowTime < nextPrayer.time) {
                            // Between this and next prayer → NOW
                            isNow = true;
                            currentPrayer = prayer;
                        } else {
                            // Past this prayer and next one already started → FINISHED
                            isFinished = true;
                        }
                    }
                    const item = document.createElement('div');
                    item.className = 'prayer-item';
                    item.innerHTML = `
        <div>
          ${prayer.name} 
          ${isNow ? '<br><span class="now-tag">Now</span>' : ''}
          ${isFinished ? '<br><span class="finished-tag">End</span>' : ''}
        </div>
        <div>${formatTime12h(prayer.timeStr24)}</div>
      `;
                    prayerList.appendChild(item);
                }
                // 🔹 Update current-prayer display
                $('current-prayer').innerHTML = currentPrayer ? `${currentPrayer.name}<br><span style="font-size:0.9em">${formatTime12h(currentPrayer.timeStr24)}</span>` : 'N/A';
                // 🔹 After rendering prayer list and current prayer
                let nextPrayer = todayPrayerTimes.find(p => p.time > nowTime);
                const nextPrayerDisplay = $('next-prayer'); // <div id="next-prayer"></div>
                const countdownEl = $('next-prayer-countdown'); // <div id="next-prayer-countdown"></div>
                if (nextPrayer) {
                    const now = new Date();
                    let diff = nextPrayer.time - now; // ms
                    if (diff < 0) diff = 0;
                    const hours = Math.floor(diff / (1000 * 60 * 60));
                    const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
                    const seconds = Math.floor((diff % (1000 * 60)) / 1000);
                    const plainName = nextPrayer.name.replace(/[\u{1F300}-\u{1FAFF}]/gu, '');
                    if (countdownEl) {
                        countdownEl.textContent = `${plainName} in ${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
                    }
                } else {
                    if (nextPrayerDisplay) {
                        nextPrayerDisplay.textContent = 'No more prayers today';
                    }
                    if (countdownEl) {
                        countdownEl.textContent = 'No more prayers today';
                    }
                }
            } catch (err) {
                console.error('fetchPrayerTimes error', err);
            }
        }
        /* Calendar UI */
        function showWeek() {
            switchTab('Bi-Weekly');
            const calendar = $('calendar');
            document.getElementById('app').style.display = 'block';
            calendar.innerHTML = '';
            const now = new Date();
            const start = new Date(now);
            start.setDate(now.getDate() - now.getDay());
            for (let i = 0; i < 14; i++) {
                const date = new Date(start);
                date.setDate(start.getDate() + i);
                const isToday = date.toDateString() === (new Date()).toDateString();
                const el = document.createElement('div');
                el.className = `calendar-day ${isToday ? 'active' : ''}`;
                el.innerHTML = `${date.toLocaleDateString('en-US', { weekday: 'short' })}<br>${date.getDate()}`;
                el.addEventListener('click', () => {
                    fetchPrayerTimes(date);
                    highlightActiveDay(el);
                });
                calendar.appendChild(el);
            }
        }

        function showMonth() {
            switchTab('Month');
            const calendar = $('calendar');
            document.getElementById('app').style.display = 'none';
            calendar.innerHTML = '';
            const now = new Date();
            const year = now.getFullYear(),
                month = now.getMonth();
            const daysInMonth = new Date(year, month + 1, 0).getDate();
            const firstDay = new Date(year, month, 1).getDay();
            for (let i = 0; i < firstDay; i++) {
                const empty = document.createElement('div');
                empty.className = 'calendar-day';
                empty.innerHTML = '&nbsp;';
                calendar.appendChild(empty);
            }
            for (let d = 1; d <= daysInMonth; d++) {
                const date = new Date(year, month, d);
                const isToday = d === now.getDate();

						const sunriseRaw = timings['Sunrise'].split(' ')[0]; // HH:MM
						const [sh, sm] = sunriseRaw.split(':').map(Number);
						const total = sh * 60 + sm + (prayer.afterMinutes || 15);
						const ih = Math.floor(total / 60) % 24;
						const im = total % 60;
						timeStr24 = `${String(ih).padStart(2, '0')}:${String(im).padStart(2, '0')}:00`;
					} else {
						if (!timings[prayer.key]) continue;
						timeStr24 = timings[prayer.key].split(' ')[0];
						if (timeStr24.length === 5) timeStr24 += ':00';
					}
					const prayerTime = getTimeObj(timeStr24, selectedDate);
					todayPrayerTimes.push({
						key: prayer.key,
						name: prayer.name,
						time: prayerTime,
						timeStr24
					});
				}
				// 🔹 Render list with NOW & FINISHED logic
				let currentPrayer = null;
				for (let i = 0; i < todayPrayerTimes.length; i++) {
					const prayer = todayPrayerTimes[i];
					const nextPrayer = todayPrayerTimes[i + 1]; // look ahead
					let isNow = false;
					let isFinished = false;
					if (nowTime >= prayer.time) {
						if (!nextPrayer || nowTime < nextPrayer.time) {
							// Between this and next prayer → NOW
							isNow = true;
							currentPrayer = prayer;
						} else {
							// Past this prayer and next one already started → FINISHED
							isFinished = true;
						}
					}
					const item = document.createElement('div');
					item.className = 'prayer-item';
					item.innerHTML = `
        <div>
          ${prayer.name} 
          ${isNow ? '<br><span class="now-tag">Now</span>' : ''}
          ${isFinished ? '<br><span class="finished-tag">End</span>' : ''}
        </div>
        <div>${formatTime12h(prayer.timeStr24)}</div>
      `;
					prayerList.appendChild(item);
				}
				// 🔹 Update current-prayer display
				$('current-prayer').innerHTML = currentPrayer ? `${currentPrayer.name}<br><span style="font-size:0.9em">${formatTime12h(currentPrayer.timeStr24)}</span>` : 'N/A';
				// 🔹 After rendering prayer list and current prayer
				let nextPrayer = todayPrayerTimes.find(p => p.time > nowTime);
				const nextPrayerDisplay = $('next-prayer'); // <div id="next-prayer"></div>
				const countdownEl = $('next-prayer-countdown'); // <div id="next-prayer-countdown"></div>
				if (nextPrayer) {
					const now = new Date();
					let diff = nextPrayer.time - now; // ms
					if (diff < 0) diff = 0;
					const hours = Math.floor(diff / (1000 * 60 * 60));
					const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
					const seconds = Math.floor((diff % (1000 * 60)) / 1000);
					const plainName = nextPrayer.name.replace(/[\u{1F300}-\u{1FAFF}]/gu, '');
					if (countdownEl) {
						countdownEl.textContent = `${plainName} in ${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
					}
				} else {
					if (nextPrayerDisplay) {
						nextPrayerDisplay.textContent = 'No more prayers today';
					}
					if (countdownEl) {
						countdownEl.textContent = 'No more prayers today';
					}
				}
			} catch (err) {
				console.error('fetchPrayerTimes error', err);
			}

		/* Format time to 12h (HH:MM AM/PM), accepts "HH:MM" or "HH:MM:SS" */
		function formatTime12h(timeStr) {
			const parts = timeStr.split(':').map(Number);
			let h = parts[0],
				m = parts[1] || 0; // ← Single || (logical OR)
			const ampm = h >= 12 ? 'PM' : 'AM';
			h = h % 12 || 12;
			return `${h}:${String(m).padStart(2, '0')} ${ampm}`;
		}

		/* Utility to parse time string "HH:MM[:SS]" into Date on given date */
		function getTimeObj(timeStr, date = selectedDateGlobal) {
			const [h, m, s = '0'] = timeStr.split(':').map(Number);
			const t = new Date(date);
			t.setHours(Number(h), Number(m), Number(s || 0), 0);
			return t;
		}

				const data = await res.json();
				$('weather-temp').textContent = `🌡 ${data.current.temperature_2m}°F`;
				$('weather-wind').textContent = `💨 ${data.current.wind_speed_10m} km/h`;

				$('weather-temp').textContent = '--';
				$('weather-wind').textContent = '--';
			}
		}

			const lon = userCoords.longitude;
			const method = 2; // you can expose this
			const day = selectedDate.getDate();
			const month = selectedDate.getMonth() + 1;
			const year = selectedDate.getFullYear();
			const url = `https://api.aladhan.com/v1/calendar?latitude=${lat}&longitude=${lon}&method=${method}&month=${month}&year=${year}&school=1`;
			try {
				const res = await fetch(url);
				const json = await res.json();
				if (!json || json.code !== 200 || !json.data) throw new Error('Aladhan API error');
				const today = json.data[day - 1];
				const timings = today.timings;
				const hijri = today.date.hijri;
				// Format Gregorian date nicely
				const dateObj = new Date(selectedDate);
				const options = {
					weekday: 'short',
					year: 'numeric',
					month: 'long',
					day: 'numeric'
				};
				const formattedDate = dateObj.toLocaleDateString('en-US', {

					year: 'numeric'
				}).replace(',', '');
				// Example output: Saturday, August 16, 2025
				$('gregorian-date').textContent = formattedDate;
				// Hijri (from API)
				$('hijri-date').textContent = `${hijri.day} ${hijri.month.en} ${hijri.year}`;
				$('hijri-date').textContent = `${hijri.day} ${hijri.month.en} ${hijri.year}`;
				// prayer order (Ishraq derived)

					name: "Imsak 🌙",

				}, {
					name: "Fajr ✨",
					key: "Fajr"
				}, {
					name: "Sunrise 🌅",
					key: "Sunrise"
				}, {
					name: "Ishraq 🌄",
					key: "Ishraq",
					derived: true,
					afterMinutes: 22
				}, {
					name: "Dhuhr ☀️",
					key: "Dhuhr"
				}, {
					name: "Asr ☁️",
					key: "Asr"
				}, {
					name: "Maghrib 🌇",
					key: "Maghrib"
				}, {
					name: "Isha’a 🌙",
					key: "Isha"
				}];
				const prayerList = $('prayer-list');
				prayerList.innerHTML = '';
				const now = new Date();
				const nowTime = new Date(now.getFullYear(), now.getMonth(), now.getDate(), now.getHours(), now.getMinutes(), now.getSeconds());
				todayPrayerTimes = []; // reset
				// Build prayer times array
				for (const prayer of prayerOrder) {
					let timeStr24;
					if (prayer.derived && prayer.key === 'Ishraq') {
						const sunriseRaw = timings['Sunrise'].split(' ')[0]; // HH:MM
						const [sh, sm] = sunriseRaw.split(':').map(Number);
						const total = sh * 60 + sm + (prayer.afterMinutes || 15);
						const ih = Math.floor(total / 60) % 24;
						const im = total % 60;
						timeStr24 = `${String(ih).padStart(2, '0')}:${String(im).padStart(2, '0')}:00`;
					} else {
						if (!timings[prayer.key]) continue;
						timeStr24 = timings[prayer.key].split(' ')[0];
						if (timeStr24.length === 5) timeStr24 += ':00';
					}
					const prayerTime = getTimeObj(timeStr24, selectedDate);
					todayPrayerTimes.push({
						key: prayer.key,
						name: prayer.name,
						time: prayerTime,
						timeStr24
					});
				}

				let currentPrayer = null;
				for (let i = 0; i < todayPrayerTimes.length; i++) {
					const prayer = todayPrayerTimes[i];
					const nextPrayer = todayPrayerTimes[i + 1]; // look ahead
					let isNow = false;
					let isFinished = false;
					if (nowTime >= prayer.time) {
						if (!nextPrayer || nowTime < nextPrayer.time) {
							// Between this and next prayer → NOW
							isNow = true;
							currentPrayer = prayer;
						} else {
							// Past this prayer and next one already started → FINISHED
							isFinished = true;
						}
					}
					const item = document.createElement('div');
					item.className = 'prayer-item';
					item.innerHTML = `
        <div>
          ${prayer.name} 
          ${isNow ? '<br><span class="now-tag">Now</span>' : ''}
          ${isFinished ? '<br><span class="finished-tag">End</span>' : ''}
        </div>
        <div>${formatTime12h(prayer.timeStr24)}</div>
      `;
					prayerList.appendChild(item);
				}
				// 🔹 Update current-prayer display
				$('current-prayer').innerHTML = currentPrayer ? `${currentPrayer.name}<br><span style="font-size:0.9em">${formatTime12h(currentPrayer.timeStr24)}</span>` : 'N/A';
				// 🔹 After rendering prayer list and current prayer
				let nextPrayer = todayPrayerTimes.find(p => p.time > nowTime);
				const nextPrayerDisplay = $('next-prayer'); // <div id="next-prayer"></div>
				const countdownEl = $('next-prayer-countdown'); // <div id="next-prayer-countdown"></div>
				if (nextPrayer) {
					const now = new Date();
					let diff = nextPrayer.time - now; // ms
					if (diff < 0) diff = 0;
					const hours = Math.floor(diff / (1000 * 60 * 60));
					const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
					const seconds = Math.floor((diff % (1000 * 60)) / 1000);
					const plainName = nextPrayer.name.replace(/[\u{1F300}-\u{1FAFF}]/gu, '');
					if (countdownEl) {
						countdownEl.textContent = `${plainName} in ${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
					}
				} else {
					if (nextPrayerDisplay) {
						nextPrayerDisplay.textContent = 'No more prayers today';
					}
					if (countdownEl) {
						countdownEl.textContent = 'No more prayers today';
					}
				}
			} catch (err) {
				console.error('fetchPrayerTimes error', err);
			}
		}

			const volume = document.getElementById("volume");
			const surahName = document.getElementById("surah-name");
			const status = document.getElementById("status");

			let currentIndex = Number(localStorage.getItem("surahIndex")) || 1;
			let isPlaying = false;
			let isUnlocked = false;

			// Use DIFFERENT variable name: quranAudio instead of audio
			quranAudio.volume = 0.4;
			quranAudio.playsInline = true;
			volume.value = 0.4;

			function loadTrack(index) {
				const num = String(index).padStart(3, "0"); // ✅ CORRECT!
				const filePath = `quran/${num}.mp3`;

				status.textContent = `🔍 ${surahList[index - 1]}...`;
				surahName.textContent = surahList[index - 1];

				// Reset position for new Surah
				localStorage.setItem("time", "0");
				progress.value = 0;
				quranAudio.currentTime = 0;

				// Just try to load - audio element handles errors gracefully
				quranAudio.src = filePath;
				quranAudio.load();
			}

			function play() {
				quranAudio.play().then(() => {
					playBtn.textContent = "⏸";
					isPlaying = true;
					status.textContent = "Playing";
				}).catch(e => {
					status.textContent = "Click to unlock";
				});
			}

			function pause() {
				quranAudio.pause();
				playBtn.textContent = "▶";
				isPlaying = false;
				status.textContent = "Paused";
			}

			playBtn.onclick = () => {
				isUnlocked = true;
				if (isPlaying) pause();
				else play();
			};

			nextBtn.onclick = () => {
				currentIndex = currentIndex % TOTAL_TRACKS + 1;
				loadTrack(currentIndex);
				localStorage.setItem("surahIndex", currentIndex);
				if (isUnlocked && isPlaying) setTimeout(play, 500);
			};

			prevBtn.onclick = () => {
				currentIndex = currentIndex === 1 ? TOTAL_TRACKS : currentIndex - 1;
				loadTrack(currentIndex);
				localStorage.setItem("surahIndex", currentIndex);
				if (isUnlocked && isPlaying) setTimeout(play, 500);
			};

			volume.oninput = () => quranAudio.volume = volume.value;

			quranAudio.ontimeupdate = () => {
				if (quranAudio.duration && quranAudio.duration > 0) {
					progress.value = (quranAudio.currentTime / quranAudio.duration) * 100;
					localStorage.setItem("time", quranAudio.currentTime);
				}
			};

			progress.oninput = () => {
				if (quranAudio.duration) {
					quranAudio.currentTime = (progress.value / 100) * quranAudio.duration;
				}
			};

			quranAudio.onloadedmetadata = () => {
				status.textContent = "Ready";
			};

			quranAudio.onended = () => {
				nextBtn.click();
			};

			quranAudio.onerror = () => {
				status.textContent = `❌ quran/${String(currentIndex).padStart(3, "0")}.mp3 not found`;
				console.error(`Quran file missing: quran/${String(currentIndex).padStart(3, "0")}.mp3`);
			};


			// Unlock Quran audio
			document.addEventListener('click', function unlockQuran() {
				if (!isUnlocked) {
					quranAudio.play().then(() => {
						quranAudio.pause();
						isUnlocked = true;
					}).catch(() => {});
					document.removeEventListener('click', unlockQuran);
				}
			}, {
				once: true
			});

			// Load initial track
			loadTrack(currentIndex);
		})();


		function refreshAllDataHourly() {
			if (!userCoords) return;

			const now = new Date();

			// Refresh prayer times for selected date (or today)
			fetchPrayerTimes(selectedDateGlobal || now);

			// Refresh weather
			fetchWeather(userCoords.latitude, userCoords.longitude);

			console.log('[Hourly Refresh]', now.toLocaleTimeString());
		}

		// Run once immediately after load
		setTimeout(refreshAllDataHourly, 3000);

		// Then every 1 hour (3600000 ms)
		setInterval(refreshAllDataHourly, 60 * 60 * 1000);

			fsBtn.textContent = document.fullscreenElement ? '⤢' : '⛶';
		});

		/* Format time to 12h (HH:MM AM/PM), accepts "HH:MM" or "HH:MM:SS" */
		function formatTime12h(timeStr) {
			const parts = timeStr.split(':').map(Number);
			let h = parts[0],
				m = parts[1] || 0; // ← Single || (logical OR)
			const ampm = h >= 12 ? 'PM' : 'AM';
			h = h % 12 || 12;
			return `${h}:${String(m).padStart(2, '0')} ${ampm}`;
		}
		/* Utility to parse time string "HH:MM[:SS]" into Date on given date */
		function getTimeObj(timeStr, date = selectedDateGlobal) {
			const [h, m, s = '0'] = timeStr.split(':').map(Number);
			const t = new Date(date);
			t.setHours(Number(h), Number(m), Number(s || 0), 0);
			return t;
		}

		/* Main: fetch prayer times from Aladhan & render */
		async function fetchPrayerTimes(selectedDate = new Date()) {
			if (!navigator.onLine) {
				console.warn('Offline – skipping prayer time fetch');
				$('current-prayer').textContent = 'Offline';
				return;
			}
			if (!userCoords) return;
			if (!userCoords) return;
			selectedDateGlobal = selectedDate;
			// Reset todayPrayerTimes
			todayPrayerTimes = [];
			azanPlayedFor = null; // reset for new day/load
			const lat = userCoords.latitude;
			const lon = userCoords.longitude;
			const method = 2; // you can expose this
			const day = selectedDate.getDate();
			const month = selectedDate.getMonth() + 1;
			const year = selectedDate.getFullYear();
			const url = `https://api.aladhan.com/v1/calendar?latitude=${lat}&longitude=${lon}&method=${method}&month=${month}&year=${year}&school=1`;
			try {
				const res = await fetch(url);
				const json = await res.json();
				if (!json || json.code !== 200 || !json.data) throw new Error('Aladhan API error');
				const today = json.data[day - 1];
				const timings = today.timings;
				const hijri = today.date.hijri;
				// Format Gregorian date nicely
				// Create date object from selected date
				const dateObj = new Date(selectedDate);
				// Format Gregorian date: Sat Feb 07, 2026
				const gregorianFormatted = dateObj.toLocaleDateString('en-US', {
					weekday: 'short',
					month: 'short',
					day: '2-digit',
					year: 'numeric'
				});
				// Format Hijri date
				const hijriFormatted = `${hijri.day} ${hijri.month.en} ${hijri.year}`;
				// Output combined result
				document.getElementById('gregorian-date').textContent =
					`${gregorianFormatted} | ${hijriFormatted}`;
				// prayer order (Ishraq derived)
				const prayerOrder = [{
					name: "Imsak",
					key: "Imsak"
				}, {
					name: "Fajr",
					key: "Fajr"
				}, {
					name: "Sunrise",
					key: "Sunrise"
				}, {
					name: "Ishraq",
					key: "Ishraq",
					derived: true,
					afterMinutes: 22
				}, {
					name: "Zuhr",
					key: "Dhuhr"
				}, {
					name: "Asr",
					key: "Asr"
				}, {
					name: "Maghrib",
					key: "Maghrib"
				}, {
					name: "Isha’a",
					key: "Isha"
				}];
				const prayerList = $('prayer-list');
				prayerList.innerHTML = '';
				const now = new Date();
				const nowTime = new Date(now.getFullYear(), now.getMonth(), now.getDate(), now.getHours(), now.getMinutes(), now.getSeconds());
				todayPrayerTimes = []; // reset
				// Build prayer times array
				for (const prayer of prayerOrder) {
					let timeStr24;
					if (prayer.derived && prayer.key === 'Ishraq') {
						const sunriseRaw = timings['Sunrise'].split(' ')[0]; // HH:MM
						const [sh, sm] = sunriseRaw.split(':').map(Number);
						const total = sh * 60 + sm + (prayer.afterMinutes || 15);
						const ih = Math.floor(total / 60) % 24;
						const im = total % 60;
						timeStr24 = `${String(ih).padStart(2, '0')}:${String(im).padStart(2, '0')}:00`;
					} else {
						if (!timings[prayer.key]) continue;
						timeStr24 = timings[prayer.key].split(' ')[0];
						if (timeStr24.length === 5) timeStr24 += ':00';
					}
					const prayerTime = getTimeObj(timeStr24, selectedDate);
					todayPrayerTimes.push({
						key: prayer.key,
						name: prayer.name,
						time: prayerTime,
						timeStr24
					});
				}
				// 🔹 Render list with NOW & FINISHED logic
				let currentPrayer = null;
				for (let i = 0; i < todayPrayerTimes.length; i++) {
					const prayer = todayPrayerTimes[i];
					const nextPrayer = todayPrayerTimes[i + 1]; // look ahead
					let isNow = false;
					let isFinished = false;
					if (nowTime >= prayer.time) {
						if (!nextPrayer || nowTime < nextPrayer.time) {
							// Between this and next prayer → NOW
							isNow = true;
							currentPrayer = prayer;
						} else {
							// Past this prayer and next one already started → FINISHED
							isFinished = true;
						}
					}
					const item = document.createElement('div');
					// Add multiple classes
					item.classList.add('prayer-item1');
					if (isNow) {
						item.setAttribute('data-set', `${prayer.name}`);
						item.classList.add('now-tag');
					} else if (isFinished) {
						item.setAttribute('data-set', `${prayer.name}`);
						item.classList.add('finished-tag');
					} else {
						item.setAttribute('data-set', prayer.name);
					}
					// Set inner content (name + formatted time)  <div class="prayer-name">${prayer.name}</div>
					item.innerHTML = `
							 
							  <div class="prayer-time">${formatTime12h(prayer.timeStr24)}</div>
							`;
					prayerList.appendChild(item);
				}
				// 🔹 Update current-prayer display
				$('current-prayer').innerHTML = currentPrayer ? `${currentPrayer.name}<br><span style="font-size:0.9em">${formatTime12h(currentPrayer.timeStr24)}</span>` : 'N/A';
				// 🔹 After rendering prayer list and current prayer
				let nextPrayer = todayPrayerTimes.find(p => p.time > nowTime);
				const nextPrayerDisplay = $('next-prayer'); // <div id="next-prayer"></div>
				const countdownEl = $('next-prayer-countdown'); // <div id="next-prayer-countdown"></div>
				if (nextPrayer) {
					const now = new Date();
					let diff = nextPrayer.time - now; // ms
					if (diff < 0) diff = 0;
					const hours = Math.floor(diff / (1000 * 60 * 60));
					const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
					const seconds = Math.floor((diff % (1000 * 60)) / 1000);
					const plainName = nextPrayer.name.replace(/[\u{1F300}-\u{1FAFF}]/gu, '');
					if (countdownEl) {
						countdownEl.textContent = `${plainName} in ${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}`;
					}
				} else {
					if (nextPrayerDisplay) {
						nextPrayerDisplay.textContent = 'All prayed out';
					}
					if (countdownEl) {
						countdownEl.textContent = 'All prayed out';
					}
				}
			} catch (err) {
				console.error('fetchPrayerTimes error', err);
			}
		}

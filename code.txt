    <script>
        /* Globals */
        let userCoords = null;
        let todayPrayerTimes = []; // { key, name, time (Date), timeStr24 }
        let azanPlayedFor = null; // key of prayer already played
        let isMuted = false;
        let selectedDateGlobal = new Date();
        const audio = document.getElementById('azan-audio');
        audio.muted = isMuted;
        // Unlock audio on first user click
        document.addEventListener('click', async () => {
            try {
                await audio.play();
                audio.pause();
                audio.currentTime = 0;
            } catch (e) {
                console.warn('Audio unlock failed', e);
            }
        });
        /* DOM helpers */
        const $ = id => document.getElementById(id);
        /* Digital clock */
        function updateDigitalClock() {
            const now = new Date();
            let hours = now.getHours();
            const minutes = now.getMinutes();
            const seconds = now.getSeconds();
            const ampm = hours >= 12 ? 'PM' : 'AM';
            hours = hours % 12 || 12;
            const timeStr = `${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
            $('digital-clock').innerHTML = `${timeStr}<br><span style="font-size:40px">${ampm}</span>`;
        }
        setInterval(updateDigitalClock, 1000);
        updateDigitalClock();
        /* Toggle mute/unmute: attempt to unlock audio on unmute (user interaction) */
        $('mute-btn').addEventListener('click', async () => {
            isMuted = !isMuted;
            audio.muted = isMuted;
            $('mute-btn').textContent = isMuted ? '🔇' : '🔊';
            if (!isMuted) {
                try {
                    await audio.play();
                    audio.pause();
                    audio.currentTime = 0;
                } catch (e) {
                    /* won't break if autoplay blocked */
                }
            }
        });
        /* Refresh button */
        $('refresh-btn').addEventListener('click', () => {
            if (userCoords) {
                fetchPrayerTimes(selectedDateGlobal);
                fetchWeather(userCoords.latitude, userCoords.longitude);
                location.reload();
            }
        });
        /* Format time to 12h (HH:MM AM/PM), accepts "HH:MM" or "HH:MM:SS" */
        function formatTime12h(timeStr) {
            const parts = timeStr.split(':').map(Number);
            let h = parts[0],
                m = parts[1] || 0;  // ← Single || (logical OR)
            const ampm = h >= 12 ? 'PM' : 'AM';
            h = h % 12 || 12;
            return `${h}:${String(m).padStart(2, '0')} ${ampm}`;
        }

        /* Utility to parse time string "HH:MM[:SS]" into Date on given date */
        function getTimeObj(timeStr, date = selectedDateGlobal) {
            const [h, m, s = '0'] = timeStr.split(':').map(Number);
            const t = new Date(date);
            t.setHours(Number(h), Number(m), Number(s || 0), 0);
            return t;
        }
        async function fetchWeather(lat, lon) {
            try {
                const url = `https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current=temperature_2m,wind_speed_10m&temperature_unit=fahrenheit`;
                const res = await fetch(url);
                const data = await res.json();
                $('weather-temp').textContent = `🌡 ${data.current.temperature_2m}°F`;
                $('weather-wind').textContent = `💨 ${data.current.wind_speed_10m} km/h`;
            } catch {
                $('weather-temp').textContent = '--';
                $('weather-wind').textContent = '--';
            }
        }
        /* Main: fetch prayer times from Aladhan & render */
        async function fetchPrayerTimes(selectedDate = new Date()) {
            if (!navigator.onLine) {
                console.warn('Offline – skipping prayer time fetch');
                $('current-prayer').textContent = 'Offline';
                return;
            }
            if (!userCoords) return;
            if (!userCoords) return;
            selectedDateGlobal = selectedDate;
            // Reset todayPrayerTimes
            todayPrayerTimes = [];
            azanPlayedFor = null; // reset for new day/load
            const lat = userCoords.latitude;
            const lon = userCoords.longitude;
            const method = 2; // you can expose this
            const day = selectedDate.getDate();
            const month = selectedDate.getMonth() + 1;
            const year = selectedDate.getFullYear();
            const url = `https://api.aladhan.com/v1/calendar?latitude=${lat}&longitude=${lon}&method=${method}&month=${month}&year=${year}&school=1`;
            try {
                const res = await fetch(url);
                const json = await res.json();
                if (!json || json.code !== 200 || !json.data) throw new Error('Aladhan API error');
                const today = json.data[day - 1];
                const timings = today.timings;
                const hijri = today.date.hijri;
                // Format Gregorian date nicely
                const dateObj = new Date(selectedDate);
                const options = {
                    weekday: 'short',
                    year: 'numeric',
                    month: 'long',
                    day: 'numeric'
                };
                const formattedDate = dateObj.toLocaleDateString('en-US', {
                    weekday: 'short',
                    month: 'short',
                    day: '2-digit',
                    year: 'numeric'
                }).replace(',', '');
                // Example output: Saturday, August 16, 2025
                $('gregorian-date').textContent = formattedDate;
                // Hijri (from API)
                $('hijri-date').textContent = `${hijri.day} ${hijri.month.en} ${hijri.year}`;
                $('hijri-date').textContent = `${hijri.day} ${hijri.month.en} ${hijri.year}`;
                // prayer order (Ishraq derived)
                const prayerOrder = [{
                    name: "Imsak 🌙",
                    key: "Imsak"
                }, {
                    name: "Fajr ✨",
                    key: "Fajr"
                }, {
                    name: "Sunrise 🌅",
                    key: "Sunrise"
                }, {
                    name: "Ishraq 🌄",
                    key: "Ishraq",
                    derived: true,
                    afterMinutes: 22
                }, {
                    name: "Dhuhr ☀️",
                    key: "Dhuhr"
                }, {
                    name: "Asr ☁️",
                    key: "Asr"
                }, {
                    name: "Maghrib 🌇",
                    key: "Maghrib"
                }, {
                    name: "Isha’a 🌙",
                    key: "Isha"
                }];
                const prayerList = $('prayer-list');
                prayerList.innerHTML = '';
                const now = new Date();
                const nowTime = new Date(now.getFullYear(), now.getMonth(), now.getDate(), now.getHours(), now.getMinutes(), now.getSeconds());
                todayPrayerTimes = []; // reset
                // Build prayer times array
                for (const prayer of prayerOrder) {
                    let timeStr24;
                    if (prayer.derived && prayer.key === 'Ishraq') {
                        const sunriseRaw = timings['Sunrise'].split(' ')[0]; // HH:MM
                        const [sh, sm] = sunriseRaw.split(':').map(Number);
                        const total = sh * 60 + sm + (prayer.afterMinutes || 15);
                        const ih = Math.floor(total / 60) % 24;
                        const im = total % 60;
                        timeStr24 = `${String(ih).padStart(2, '0')}:${String(im).padStart(2, '0')}:00`;
                    } else {
                        if (!timings[prayer.key]) continue;
                        timeStr24 = timings[prayer.key].split(' ')[0];
                        if (timeStr24.length === 5) timeStr24 += ':00';
                    }
                    const prayerTime = getTimeObj(timeStr24, selectedDate);
                    todayPrayerTimes.push({
                        key: prayer.key,
                        name: prayer.name,
                        time: prayerTime,
                        timeStr24
                    });
                }
                // 🔹 Render list with NOW & FINISHED logic
                let currentPrayer = null;
                for (let i = 0; i < todayPrayerTimes.length; i++) {
                    const prayer = todayPrayerTimes[i];
                    const nextPrayer = todayPrayerTimes[i + 1]; // look ahead
                    let isNow = false;
                    let isFinished = false;
                    if (nowTime >= prayer.time) {
                        if (!nextPrayer || nowTime < nextPrayer.time) {
                            // Between this and next prayer → NOW
                            isNow = true;
                            currentPrayer = prayer;
                        } else {
                            // Past this prayer and next one already started → FINISHED
                            isFinished = true;
                        }
                    }
                    const item = document.createElement('div');
                    item.className = 'prayer-item';
                    item.innerHTML = `
        <div>
          ${prayer.name} 
          ${isNow ? '<br><span class="now-tag">Now</span>' : ''}
          ${isFinished ? '<br><span class="finished-tag">End</span>' : ''}
        </div>
        <div>${formatTime12h(prayer.timeStr24)}</div>
      `;
                    prayerList.appendChild(item);
                }
                // 🔹 Update current-prayer display
                $('current-prayer').innerHTML = currentPrayer ? `${currentPrayer.name}<br><span style="font-size:0.9em">${formatTime12h(currentPrayer.timeStr24)}</span>` : 'N/A';
                // 🔹 After rendering prayer list and current prayer
                let nextPrayer = todayPrayerTimes.find(p => p.time > nowTime);
                const nextPrayerDisplay = $('next-prayer'); // <div id="next-prayer"></div>
                const countdownEl = $('next-prayer-countdown'); // <div id="next-prayer-countdown"></div>
                if (nextPrayer) {
                    const now = new Date();
                    let diff = nextPrayer.time - now; // ms
                    if (diff < 0) diff = 0;
                    const hours = Math.floor(diff / (1000 * 60 * 60));
                    const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
                    const seconds = Math.floor((diff % (1000 * 60)) / 1000);
                    const plainName = nextPrayer.name.replace(/[\u{1F300}-\u{1FAFF}]/gu, '');
                    if (countdownEl) {
                        countdownEl.textContent = `${plainName} in ${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
                    }
                } else {
                    if (nextPrayerDisplay) {
                        nextPrayerDisplay.textContent = 'No more prayers today';
                    }
                    if (countdownEl) {
                        countdownEl.textContent = 'No more prayers today';
                    }
                }
            } catch (err) {
                console.error('fetchPrayerTimes error', err);
            }
        }
        /* Calendar UI */
        function showWeek() {
            switchTab('Bi-Weekly');
            const calendar = $('calendar');
            document.getElementById('app').style.display = 'block';
            calendar.innerHTML = '';
            const now = new Date();
            const start = new Date(now);
            start.setDate(now.getDate() - now.getDay());
            for (let i = 0; i < 14; i++) {
                const date = new Date(start);
                date.setDate(start.getDate() + i);
                const isToday = date.toDateString() === (new Date()).toDateString();
                const el = document.createElement('div');
                el.className = `calendar-day ${isToday ? 'active' : ''}`;
                el.innerHTML = `${date.toLocaleDateString('en-US', { weekday: 'short' })}<br>${date.getDate()}`;
                el.addEventListener('click', () => {
                    fetchPrayerTimes(date);
                    highlightActiveDay(el);
                });
                calendar.appendChild(el);
            }
        }

        function showMonth() {
            switchTab('Month');
            const calendar = $('calendar');
            document.getElementById('app').style.display = 'none';
            calendar.innerHTML = '';
            const now = new Date();
            const year = now.getFullYear(),
                month = now.getMonth();
            const daysInMonth = new Date(year, month + 1, 0).getDate();
            const firstDay = new Date(year, month, 1).getDay();
            for (let i = 0; i < firstDay; i++) {
                const empty = document.createElement('div');
                empty.className = 'calendar-day';
                empty.innerHTML = '&nbsp;';
                calendar.appendChild(empty);
            }
            for (let d = 1; d <= daysInMonth; d++) {
                const date = new Date(year, month, d);
                const isToday = d === now.getDate();

						const sunriseRaw = timings['Sunrise'].split(' ')[0]; // HH:MM
						const [sh, sm] = sunriseRaw.split(':').map(Number);
						const total = sh * 60 + sm + (prayer.afterMinutes || 15);
						const ih = Math.floor(total / 60) % 24;
						const im = total % 60;
						timeStr24 = `${String(ih).padStart(2, '0')}:${String(im).padStart(2, '0')}:00`;
					} else {
						if (!timings[prayer.key]) continue;
						timeStr24 = timings[prayer.key].split(' ')[0];
						if (timeStr24.length === 5) timeStr24 += ':00';
					}
					const prayerTime = getTimeObj(timeStr24, selectedDate);
					todayPrayerTimes.push({
						key: prayer.key,
						name: prayer.name,
						time: prayerTime,
						timeStr24
					});
				}
				// 🔹 Render list with NOW & FINISHED logic
				let currentPrayer = null;
				for (let i = 0; i < todayPrayerTimes.length; i++) {
					const prayer = todayPrayerTimes[i];
					const nextPrayer = todayPrayerTimes[i + 1]; // look ahead
					let isNow = false;
					let isFinished = false;
					if (nowTime >= prayer.time) {
						if (!nextPrayer || nowTime < nextPrayer.time) {
							// Between this and next prayer → NOW
							isNow = true;
							currentPrayer = prayer;
						} else {
							// Past this prayer and next one already started → FINISHED
							isFinished = true;
						}
					}
					const item = document.createElement('div');
					item.className = 'prayer-item';
					item.innerHTML = `
        <div>
          ${prayer.name} 
          ${isNow ? '<br><span class="now-tag">Now</span>' : ''}
          ${isFinished ? '<br><span class="finished-tag">End</span>' : ''}
        </div>
        <div>${formatTime12h(prayer.timeStr24)}</div>
      `;
					prayerList.appendChild(item);
				}
				// 🔹 Update current-prayer display
				$('current-prayer').innerHTML = currentPrayer ? `${currentPrayer.name}<br><span style="font-size:0.9em">${formatTime12h(currentPrayer.timeStr24)}</span>` : 'N/A';
				// 🔹 After rendering prayer list and current prayer
				let nextPrayer = todayPrayerTimes.find(p => p.time > nowTime);
				const nextPrayerDisplay = $('next-prayer'); // <div id="next-prayer"></div>
				const countdownEl = $('next-prayer-countdown'); // <div id="next-prayer-countdown"></div>
				if (nextPrayer) {
					const now = new Date();
					let diff = nextPrayer.time - now; // ms
					if (diff < 0) diff = 0;
					const hours = Math.floor(diff / (1000 * 60 * 60));
					const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
					const seconds = Math.floor((diff % (1000 * 60)) / 1000);
					const plainName = nextPrayer.name.replace(/[\u{1F300}-\u{1FAFF}]/gu, '');
					if (countdownEl) {
						countdownEl.textContent = `${plainName} in ${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
					}
				} else {
					if (nextPrayerDisplay) {
						nextPrayerDisplay.textContent = 'No more prayers today';
					}
					if (countdownEl) {
						countdownEl.textContent = 'No more prayers today';
					}
				}
			} catch (err) {
				console.error('fetchPrayerTimes error', err);
			}

		/* Format time to 12h (HH:MM AM/PM), accepts "HH:MM" or "HH:MM:SS" */
		function formatTime12h(timeStr) {
			const parts = timeStr.split(':').map(Number);
			let h = parts[0],
				m = parts[1] || 0; // ← Single || (logical OR)
			const ampm = h >= 12 ? 'PM' : 'AM';
			h = h % 12 || 12;
			return `${h}:${String(m).padStart(2, '0')} ${ampm}`;
		}

		/* Utility to parse time string "HH:MM[:SS]" into Date on given date */
		function getTimeObj(timeStr, date = selectedDateGlobal) {
			const [h, m, s = '0'] = timeStr.split(':').map(Number);
			const t = new Date(date);
			t.setHours(Number(h), Number(m), Number(s || 0), 0);
			return t;
		}

				const data = await res.json();
				$('weather-temp').textContent = `🌡 ${data.current.temperature_2m}°F`;
				$('weather-wind').textContent = `💨 ${data.current.wind_speed_10m} km/h`;

				$('weather-temp').textContent = '--';
				$('weather-wind').textContent = '--';
			}
		}

			const lon = userCoords.longitude;
			const method = 2; // you can expose this
			const day = selectedDate.getDate();
			const month = selectedDate.getMonth() + 1;
			const year = selectedDate.getFullYear();
			const url = `https://api.aladhan.com/v1/calendar?latitude=${lat}&longitude=${lon}&method=${method}&month=${month}&year=${year}&school=1`;
			try {
				const res = await fetch(url);
				const json = await res.json();
				if (!json || json.code !== 200 || !json.data) throw new Error('Aladhan API error');
				const today = json.data[day - 1];
				const timings = today.timings;
				const hijri = today.date.hijri;
				// Format Gregorian date nicely
				const dateObj = new Date(selectedDate);
				const options = {
					weekday: 'short',
					year: 'numeric',
					month: 'long',
					day: 'numeric'
				};
				const formattedDate = dateObj.toLocaleDateString('en-US', {

					year: 'numeric'
				}).replace(',', '');
				// Example output: Saturday, August 16, 2025
				$('gregorian-date').textContent = formattedDate;
				// Hijri (from API)
				$('hijri-date').textContent = `${hijri.day} ${hijri.month.en} ${hijri.year}`;
				$('hijri-date').textContent = `${hijri.day} ${hijri.month.en} ${hijri.year}`;
				// prayer order (Ishraq derived)

					name: "Imsak 🌙",

				}, {
					name: "Fajr ✨",
					key: "Fajr"
				}, {
					name: "Sunrise 🌅",
					key: "Sunrise"
				}, {
					name: "Ishraq 🌄",
					key: "Ishraq",
					derived: true,
					afterMinutes: 22
				}, {
					name: "Dhuhr ☀️",
					key: "Dhuhr"
				}, {
					name: "Asr ☁️",
					key: "Asr"
				}, {
					name: "Maghrib 🌇",
					key: "Maghrib"
				}, {
					name: "Isha’a 🌙",
					key: "Isha"
				}];
				const prayerList = $('prayer-list');
				prayerList.innerHTML = '';
				const now = new Date();
				const nowTime = new Date(now.getFullYear(), now.getMonth(), now.getDate(), now.getHours(), now.getMinutes(), now.getSeconds());
				todayPrayerTimes = []; // reset
				// Build prayer times array
				for (const prayer of prayerOrder) {
					let timeStr24;
					if (prayer.derived && prayer.key === 'Ishraq') {
						const sunriseRaw = timings['Sunrise'].split(' ')[0]; // HH:MM
						const [sh, sm] = sunriseRaw.split(':').map(Number);
						const total = sh * 60 + sm + (prayer.afterMinutes || 15);
						const ih = Math.floor(total / 60) % 24;
						const im = total % 60;
						timeStr24 = `${String(ih).padStart(2, '0')}:${String(im).padStart(2, '0')}:00`;
					} else {
						if (!timings[prayer.key]) continue;
						timeStr24 = timings[prayer.key].split(' ')[0];
						if (timeStr24.length === 5) timeStr24 += ':00';
					}
					const prayerTime = getTimeObj(timeStr24, selectedDate);
					todayPrayerTimes.push({
						key: prayer.key,
						name: prayer.name,
						time: prayerTime,
						timeStr24
					});
				}

				let currentPrayer = null;
				for (let i = 0; i < todayPrayerTimes.length; i++) {
					const prayer = todayPrayerTimes[i];
					const nextPrayer = todayPrayerTimes[i + 1]; // look ahead
					let isNow = false;
					let isFinished = false;
					if (nowTime >= prayer.time) {
						if (!nextPrayer || nowTime < nextPrayer.time) {
							// Between this and next prayer → NOW
							isNow = true;
							currentPrayer = prayer;
						} else {
							// Past this prayer and next one already started → FINISHED
							isFinished = true;
						}
					}
					const item = document.createElement('div');
					item.className = 'prayer-item';
					item.innerHTML = `
        <div>
          ${prayer.name} 
          ${isNow ? '<br><span class="now-tag">Now</span>' : ''}
          ${isFinished ? '<br><span class="finished-tag">End</span>' : ''}
        </div>
        <div>${formatTime12h(prayer.timeStr24)}</div>
      `;
					prayerList.appendChild(item);
				}
				// 🔹 Update current-prayer display
				$('current-prayer').innerHTML = currentPrayer ? `${currentPrayer.name}<br><span style="font-size:0.9em">${formatTime12h(currentPrayer.timeStr24)}</span>` : 'N/A';
				// 🔹 After rendering prayer list and current prayer
				let nextPrayer = todayPrayerTimes.find(p => p.time > nowTime);
				const nextPrayerDisplay = $('next-prayer'); // <div id="next-prayer"></div>
				const countdownEl = $('next-prayer-countdown'); // <div id="next-prayer-countdown"></div>
				if (nextPrayer) {
					const now = new Date();
					let diff = nextPrayer.time - now; // ms
					if (diff < 0) diff = 0;
					const hours = Math.floor(diff / (1000 * 60 * 60));
					const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
					const seconds = Math.floor((diff % (1000 * 60)) / 1000);
					const plainName = nextPrayer.name.replace(/[\u{1F300}-\u{1FAFF}]/gu, '');
					if (countdownEl) {
						countdownEl.textContent = `${plainName} in ${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
					}
				} else {
					if (nextPrayerDisplay) {
						nextPrayerDisplay.textContent = 'No more prayers today';
					}
					if (countdownEl) {
						countdownEl.textContent = 'No more prayers today';
					}
				}
			} catch (err) {
				console.error('fetchPrayerTimes error', err);
			}
		}

			const volume = document.getElementById("volume");
			const surahName = document.getElementById("surah-name");
			const status = document.getElementById("status");

			let currentIndex = Number(localStorage.getItem("surahIndex")) || 1;
			let isPlaying = false;
			let isUnlocked = false;

			// Use DIFFERENT variable name: quranAudio instead of audio
			quranAudio.volume = 0.4;
			quranAudio.playsInline = true;
			volume.value = 0.4;

			function loadTrack(index) {
				const num = String(index).padStart(3, "0"); // ✅ CORRECT!
				const filePath = `quran/${num}.mp3`;

				status.textContent = `🔍 ${surahList[index - 1]}...`;
				surahName.textContent = surahList[index - 1];

				// Reset position for new Surah
				localStorage.setItem("time", "0");
				progress.value = 0;
				quranAudio.currentTime = 0;

				// Just try to load - audio element handles errors gracefully
				quranAudio.src = filePath;
				quranAudio.load();
			}

			function play() {
				quranAudio.play().then(() => {
					playBtn.textContent = "⏸";
					isPlaying = true;
					status.textContent = "Playing";
				}).catch(e => {
					status.textContent = "Click to unlock";
				});
			}

			function pause() {
				quranAudio.pause();
				playBtn.textContent = "▶";
				isPlaying = false;
				status.textContent = "Paused";
			}

			playBtn.onclick = () => {
				isUnlocked = true;
				if (isPlaying) pause();
				else play();
			};

			nextBtn.onclick = () => {
				currentIndex = currentIndex % TOTAL_TRACKS + 1;
				loadTrack(currentIndex);
				localStorage.setItem("surahIndex", currentIndex);
				if (isUnlocked && isPlaying) setTimeout(play, 500);
			};

			prevBtn.onclick = () => {
				currentIndex = currentIndex === 1 ? TOTAL_TRACKS : currentIndex - 1;
				loadTrack(currentIndex);
				localStorage.setItem("surahIndex", currentIndex);
				if (isUnlocked && isPlaying) setTimeout(play, 500);
			};

			volume.oninput = () => quranAudio.volume = volume.value;

			quranAudio.ontimeupdate = () => {
				if (quranAudio.duration && quranAudio.duration > 0) {
					progress.value = (quranAudio.currentTime / quranAudio.duration) * 100;
					localStorage.setItem("time", quranAudio.currentTime);
				}
			};

			progress.oninput = () => {
				if (quranAudio.duration) {
					quranAudio.currentTime = (progress.value / 100) * quranAudio.duration;
				}
			};

			quranAudio.onloadedmetadata = () => {
				status.textContent = "Ready";
			};

			quranAudio.onended = () => {
				nextBtn.click();
			};

			quranAudio.onerror = () => {
				status.textContent = `❌ quran/${String(currentIndex).padStart(3, "0")}.mp3 not found`;
				console.error(`Quran file missing: quran/${String(currentIndex).padStart(3, "0")}.mp3`);
			};


			// Unlock Quran audio
			document.addEventListener('click', function unlockQuran() {
				if (!isUnlocked) {
					quranAudio.play().then(() => {
						quranAudio.pause();
						isUnlocked = true;
					}).catch(() => {});
					document.removeEventListener('click', unlockQuran);
				}
			}, {
				once: true
			});

			// Load initial track
			loadTrack(currentIndex);
		})();


		function refreshAllDataHourly() {
			if (!userCoords) return;

			const now = new Date();

			// Refresh prayer times for selected date (or today)
			fetchPrayerTimes(selectedDateGlobal || now);

			// Refresh weather
			fetchWeather(userCoords.latitude, userCoords.longitude);

			console.log('[Hourly Refresh]', now.toLocaleTimeString());
		}

		// Run once immediately after load
		setTimeout(refreshAllDataHourly, 3000);

		// Then every 1 hour (3600000 ms)
		setInterval(refreshAllDataHourly, 60 * 60 * 1000);

			fsBtn.textContent = document.fullscreenElement ? '⤢' : '⛶';
		});

		/* Format time to 12h (HH:MM AM/PM), accepts "HH:MM" or "HH:MM:SS" */
		function formatTime12h(timeStr) {
			const parts = timeStr.split(':').map(Number);
			let h = parts[0],
				m = parts[1] || 0; // ← Single || (logical OR)
			const ampm = h >= 12 ? 'PM' : 'AM';
			h = h % 12 || 12;
			return `${h}:${String(m).padStart(2, '0')} ${ampm}`;
		}
		/* Utility to parse time string "HH:MM[:SS]" into Date on given date */
		function getTimeObj(timeStr, date = selectedDateGlobal) {
			const [h, m, s = '0'] = timeStr.split(':').map(Number);
			const t = new Date(date);
			t.setHours(Number(h), Number(m), Number(s || 0), 0);
			return t;

			if (!navigator.onLine) {

			todayPrayerTimes = [];
			azanPlayedFor = null; // reset for new day/load

			const lon = userCoords.longitude;
			const method = 2; // you can expose this
			const day = selectedDate.getDate();
			const month = selectedDate.getMonth() + 1;
			const year = selectedDate.getFullYear();
			const url = `https://api.aladhan.com/v1/calendar?latitude=${lat}&longitude=${lon}&method=${method}&month=${month}&year=${year}&school=1`;
			try {
				const res = await fetch(url);
				const json = await res.json();
				if (!json || json.code !== 200 || !json.data) throw new Error('Aladhan API error');
				const today = json.data[day - 1];
				const timings = today.timings;
				const hijri = today.date.hijri;
				// Format Gregorian date nicely
				// Create date object from selected date
				const dateObj = new Date(selectedDate);

				const gregorianFormatted = dateObj.toLocaleDateString('en-US', {

					year: 'numeric'

					name: "Sunrise",
					key: "Sunrise"
				}, {
					name: "Ishraq",
					key: "Ishraq",
					derived: true,
					afterMinutes: 22
				}, {
					name: "Zuhr",
					key: "Dhuhr"
				}, {
					name: "Asr",
					key: "Asr"
				}, {
					name: "Maghrib",
					key: "Maghrib"
				}, {
					name: "Isha’a",
					key: "Isha"
				}];

				prayerList.innerHTML = '';
				const now = new Date();
				const nowTime = new Date(now.getFullYear(), now.getMonth(), now.getDate(), now.getHours(), now.getMinutes(), now.getSeconds());
				todayPrayerTimes = []; // reset
				// Build prayer times array
				for (const prayer of prayerOrder) {
					let timeStr24;
					if (prayer.derived && prayer.key === 'Ishraq') {
						const sunriseRaw = timings['Sunrise'].split(' ')[0]; // HH:MM
						const [sh, sm] = sunriseRaw.split(':').map(Number);
						const total = sh * 60 + sm + (prayer.afterMinutes || 15);
						const ih = Math.floor(total / 60) % 24;
						const im = total % 60;
						timeStr24 = `${String(ih).padStart(2, '0')}:${String(im).padStart(2, '0')}:00`;
					} else {
						if (!timings[prayer.key]) continue;
						timeStr24 = timings[prayer.key].split(' ')[0];
						if (timeStr24.length === 5) timeStr24 += ':00';
					}
					const prayerTime = getTimeObj(timeStr24, selectedDate);
					todayPrayerTimes.push({
						key: prayer.key,
						name: prayer.name,
						time: prayerTime,
						timeStr24
					});

				let currentPrayer = null;
				for (let i = 0; i < todayPrayerTimes.length; i++) {
					const prayer = todayPrayerTimes[i];
					const nextPrayer = todayPrayerTimes[i + 1]; // look ahead
					let isNow = false;
					let isFinished = false;
					if (nowTime >= prayer.time) {
						if (!nextPrayer || nowTime < nextPrayer.time) {
							// Between this and next prayer → NOW
							isNow = true;
							currentPrayer = prayer;
						} else {
							// Past this prayer and next one already started → FINISHED
							isFinished = true;
						}
					}
					const item = document.createElement('div');
					// Add multiple classes
					item.classList.add('prayer-item1');
					if (isNow) {
						item.setAttribute('data-set', `${prayer.name}`);
						item.classList.add('now-tag');
					} else if (isFinished) {
						item.setAttribute('data-set', `${prayer.name}`);
						item.classList.add('finished-tag');
					} else {
						item.setAttribute('data-set', prayer.name);
					}
					// Set inner content (name + formatted time)  <div class="prayer-name">${prayer.name}</div>
					item.innerHTML = `
							 
							  <div class="prayer-time">${formatTime12h(prayer.timeStr24)}</div>
							`;
					prayerList.appendChild(item);
				}
				// 🔹 Update current-prayer display
				$('current-prayer').innerHTML = currentPrayer ? `${currentPrayer.name}<br><span style="font-size:0.9em">${formatTime12h(currentPrayer.timeStr24)}</span>` : 'N/A';

				let nextPrayer = todayPrayerTimes.find(p => p.time > nowTime);
				const nextPrayerDisplay = $('next-prayer'); // <div id="next-prayer"></div>
				const countdownEl = $('next-prayer-countdown'); // <div id="next-prayer-countdown"></div>
				if (nextPrayer) {
					const now = new Date();
					let diff = nextPrayer.time - now; // ms
					if (diff < 0) diff = 0;
					const hours = Math.floor(diff / (1000 * 60 * 60));
					const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
					const seconds = Math.floor((diff % (1000 * 60)) / 1000);
					const plainName = nextPrayer.name.replace(/[\u{1F300}-\u{1FAFF}]/gu, '');
					if (countdownEl) {
						countdownEl.textContent = `${plainName} in ${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}`;
					}
				} else {
					if (nextPrayerDisplay) {
						nextPrayerDisplay.textContent = 'All prayed out';
					}
					if (countdownEl) {
						countdownEl.textContent = 'All prayed out';
					}
				}
			} catch (err) {
				console.error('fetchPrayerTimes error', err);
			}

		function fmt12(timeStr) {

			const hh = h % 12 || 12;
			return {
				time: `${hh}:${String(m).padStart(2,'0')}`,

			const {
				latitude: lat,
				longitude: lon
			} = userCoords;
			const d = date.getDate(),
				mo = date.getMonth() + 1,
				yr = date.getFullYear();
			const url = `https://api.aladhan.com/v1/calendar?latitude=${lat}&longitude=${lon}&method=2&month=${mo}&year=${yr}&school=1`;

			try {
				const res = await fetch(url);
				const json = await res.json();
				if (json.code !== 200) throw new Error();
				const today = json.data[d - 1];
				const timings = today.timings;
				const hijri = today.date.hijri;

				$('greg-date').textContent = date.toLocaleDateString('en-US', {
					weekday: 'short',
					month: 'short',
					day: '2-digit',
					year: 'numeric'
				}).toUpperCase();
				$('hijri-date').textContent = `${hijri.day} ${hijri.month.en} ${hijri.year}`;

				const order = [{
						name: 'Imsak',
						key: 'Imsak'
					},
					{
						name: 'Fajr',
						key: 'Fajr'
					},
					{
						name: 'Sunrise',
						key: 'Sunrise'
					},
					{
						name: 'Ishraq',
						key: 'Ishraq',
						derived: true,
						after: 22
					},
					{
						name: 'Zuhr',
						key: 'Dhuhr'
					},
					{
						name: 'Asr',
						key: 'Asr'
					},
					{
						name: 'Maghrib',
						key: 'Maghrib'
					},
					{
						name: "Isha'a",
						key: 'Isha'
					},
				];

				for (const p of order) {
					let ts;
					if (p.derived) {
						const raw = timings['Sunrise'].split(' ')[0];
						const [sh, sm] = raw.split(':').map(Number);
						const tot = sh * 60 + sm + p.after;
						ts = `${String(Math.floor(tot/60)%24).padStart(2,'0')}:${String(tot%60).padStart(2,'0')}:00`;
					} else {
						if (!timings[p.key]) continue;
						ts = timings[p.key].split(' ')[0];
						if (ts.length === 5) ts += ':00';
					}
					todayPrayerTimes.push({
						key: p.key,
						name: p.name,
						time: getTimeObj(ts, date),
						ts
					});
				}

				renderPrayers();
				updateCurrentPrayer();
				startCountdown();
			} catch (e) {
				console.error(e);
			}
		}

		function renderPrayers() {
			const now = new Date();
			const grid = $('prayer-grid');
			grid.innerHTML = '';

			let currentPrayer = null;

			for (let i = 0; i < todayPrayerTimes.length; i++) {
				const p = todayPrayerTimes[i];
				const next = todayPrayerTimes[i + 1];
				const isNow = now >= p.time && (!next || now < next.time);
				const isDone = !isNow && now >= p.time;
				if (isNow) currentPrayer = p;

				const {
					time,
					ampm
				} = fmt12(p.ts);
				const card = document.createElement('div');
				card.className = `prayer-card${isNow?' now':isDone?' done':''}`;
				card.innerHTML = `
      <div class="pc-name">${p.name}</div>
      <div class="pc-time">${time} <span class="pc-ampm">${ampm}</span></div>
      ${isNow ? `<div class="pc-tag now-pill">now</div>` : isDone ? `<div class="pc-tag done-pill">✓ done</div>` : ''}
    `;
				grid.appendChild(card);
			}

			// Update current block
			if (currentPrayer) {
				$('current-name').textContent = currentPrayer.name;
				const {
					time,
					ampm
				} = fmt12(currentPrayer.ts);
				$('current-time').textContent = `${time} ${ampm}`;
			} else {
				$('current-name').textContent = '—';
				$('current-time').textContent = '—';
			}
		}

		function updateCurrentPrayer() {
			const now = new Date();
			let current = null;
			for (let i = 0; i < todayPrayerTimes.length; i++) {
				const p = todayPrayerTimes[i];
				const next = todayPrayerTimes[i + 1];
				if (now >= p.time && (!next || now < next.time)) {
					current = p;
					break;
				}
			}
			if (current) {
				$('current-name').textContent = current.name;
				const {
					time,
					ampm
				} = fmt12(current.ts);
				$('current-time').textContent = `${time} ${ampm}`;
			}
		}

		let countdownTimer = null;

		function startCountdown() {
			if (countdownTimer) clearInterval(countdownTimer);
			countdownTimer = setInterval(() => {
				const now = new Date();
				const next = todayPrayerTimes.find(p => p.time > now);
				if (next) {
					let diff = next.time - now;
					const h = Math.floor(diff / 3600000);
					const m = Math.floor((diff % 3600000) / 60000);
					const s = Math.floor((diff % 60000) / 1000);
					$('next-countdown').textContent = `${next.name}  ${String(h).padStart(2,'0')}:${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
				} else {
					$('next-countdown').textContent = 'All done ✦';
				}
			}, 1000);
		}

		// ─── AZAN ───
		setInterval(() => {
			if (!todayPrayerTimes.length || isMuted) return;
			const now = new Date();
			for (const p of todayPrayerTimes) {
				const diff = now - p.time;
				if (diff >= 0 && diff < 19000 && azanPlayedFor !== p.key) {
					azanAudio.currentTime = 0;
					azanAudio.play().catch(() => {});
					azanPlayedFor = p.key;
				}
			}
		}, 1000);

		// ─── CALENDAR ───
		function showWeek() {
			$('tab-week').classList.add('active');
			$('tab-month').classList.remove('active');
			const cal = $('calendar');
			cal.innerHTML = '';
			const now = new Date();
			const start = new Date(now);
			start.setDate(now.getDate() - now.getDay());

			for (let i = 0; i < 14; i++) {

				date.setDate(start.getDate() + i);

				const el = document.createElement('div');
				el.className = `cal-day${isToday?' active':''}`;
				el.innerHTML = `<span class="cal-wday">${date.toLocaleDateString('en-US',{weekday:'short'})}</span><span class="cal-num">${date.getDate()}</span>`;
				el.addEventListener('click', () => {
					document.querySelectorAll('.cal-day').forEach(e => e.classList.remove('active'));
					el.classList.add('active');
					fetchPrayerTimes(date);
				});
				cal.appendChild(el);
			}
		}

		function showMonth() {
			$('tab-week').classList.remove('active');
			$('tab-month').classList.add('active');
			const cal = $('calendar');
			cal.innerHTML = '';
			const now = new Date();
			const year = now.getFullYear(),
				month = now.getMonth();
			const days = new Date(year, month + 1, 0).getDate();
			const first = new Date(year, month, 1).getDay();

			for (let i = 0; i < first; i++) {
				const empty = document.createElement('div');
				empty.className = 'cal-day';
				empty.style.visibility = 'hidden';
				cal.appendChild(empty);
			}
			for (let d = 1; d <= days; d++) {
				const date = new Date(year, month, d);
				const isToday = d === now.getDate();
				const el = document.createElement('div');
				el.className = `cal-day${isToday?' active':''}`;
				el.innerHTML = `<span class="cal-wday">${date.toLocaleDateString('en-US',{weekday:'short'})}</span><span class="cal-num">${d}</span>`;
				el.addEventListener('click', () => {
					document.querySelectorAll('.cal-day').forEach(e => e.classList.remove('active'));
					el.classList.add('active');
					fetchPrayerTimes(date);
				});
				cal.appendChild(el);
			}
		}

				once: true

			const ap = h >= 12 ? 'PM' : 'AM';
			h = h % 12 || 12;

			$('cl-m').textContent = String(m).padStart(2, '0');
			$('cl-s').textContent = String(s).padStart(2, '0');
			$('cl-ap').textContent = ap;

		function fmt12(ts) {
			const [h, m] = ts.split(':').map(Number);
			const ap = h >= 12 ? 'PM' : 'AM';
			return {
				time: `${(h%12||12)}:${String(m).padStart(2,'0')}`,
				ampm: ap
			};

				latitude: lat,
				longitude: lon
			} = userCoords;
			const d = date.getDate(),
				mo = date.getMonth() + 1,
				yr = date.getFullYear();
			try {
				const res = await fetch(`https://api.aladhan.com/v1/calendar?latitude=${lat}&longitude=${lon}&method=2&month=${mo}&year=${yr}&school=1`);
				const json = await res.json();
				if (json.code !== 200) throw new Error();
				const day = json.data[d - 1];
				const T = day.timings,
					H = day.date.hijri;

				$('greg-date').textContent = date.toLocaleDateString('en-US', {
					weekday: 'short',
					month: 'short',
					day: '2-digit',
					year: 'numeric'
				}).toUpperCase();
				$('hijri-date').textContent = `${H.day} ${H.month.en} ${H.year}`;

				const order = [{
						name: 'Imsak',
						key: 'Imsak'
					}, {
						name: 'Fajr',
						key: 'Fajr'
					}, {
						name: 'Sunrise',
						key: 'Sunrise'
					},
					{
						name: 'Ishraq',
						key: 'Ishraq',
						derived: true,
						after: 22
					},
					{
						name: 'Zuhr',
						key: 'Dhuhr'
					}, {
						name: 'Asr',
						key: 'Asr'
					}, {
						name: 'Maghrib',
						key: 'Maghrib'
					}, {
						name: "Isha'a",
						key: 'Isha'
					}
				];
				for (const p of order) {
					let ts;
					if (p.derived) {
						const [sh, sm] = T['Sunrise'].split(' ')[0].split(':').map(Number);
						const tot = sh * 60 + sm + p.after;
						ts = `${String(Math.floor(tot/60)%24).padStart(2,'0')}:${String(tot%60).padStart(2,'0')}:00`;
					} else {
						if (!T[p.key]) continue;
						ts = T[p.key].split(' ')[0];
						if (ts.length === 5) ts += ':00';
					}
					todayPrayers.push({
						key: p.key,
						name: p.name,
						time: toDate(ts, date),
						ts
					});
				}
				renderGrid();
				startCountdown();
			} catch (e) {
				console.error(e);
			}
		}

		function renderGrid() {
			const now = new Date(),
				grid = $('prayer-grid');
			grid.innerHTML = '';
			let cur = null;
			for (let i = 0; i < todayPrayers.length; i++) {
				const p = todayPrayers[i],
					nx = todayPrayers[i + 1];
				const isNow = now >= p.time && (!nx || now < nx.time);
				const isDone = !isNow && now >= p.time;
				if (isNow) cur = p;
				const {
					time,
					ampm
				} = fmt12(p.ts);
				const card = document.createElement('div');
				card.className = `prayer-card${isNow?' now':isDone?' done':''}`;
				card.innerHTML = `
      <div class="pc-name">${p.name}</div>
      <div class="pc-time">${time} <span class="pc-ampm">${ampm}</span></div>
      ${isNow?`<span class="pc-pill now-pill">now</span>`:isDone?`<span class="pc-pill done-pill">✓ done</span>`:''}
    `;
				grid.appendChild(card);
			}
			if (cur) {
				$('cur-name').textContent = cur.name;
				const {
					time,
					ampm
				} = fmt12(cur.ts);
				$('cur-time').textContent = `${time} ${ampm}`;
			} else {
				$('cur-name').textContent = '—';
				$('cur-time').textContent = '—';
			}
		}

		let cdTimer = null;

		function startCountdown() {
			if (cdTimer) clearInterval(cdTimer);
			cdTimer = setInterval(() => {
				const now = new Date(),
					nx = todayPrayers.find(p => p.time > now);
				if (nx) {
					const d = nx.time - now;
					const h = Math.floor(d / 3600000),
						m = Math.floor((d % 3600000) / 60000),
						s = Math.floor((d % 60000) / 1000);
					$('next-cd').textContent = `${nx.name} ${String(h).padStart(2,'0')}:${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
				} else $('next-cd').textContent = 'All done ✦';
			}, 1000);
		}

		// ─── AZAN ───
		setInterval(() => {
			if (!todayPrayers.length || isMuted) return;
			const now = new Date();
			for (const p of todayPrayers) {
				const diff = now - p.time;
				if (diff >= 0 && diff < 19000 && azanPlayed !== p.key) {
					azanAudio.currentTime = 0;
					azanAudio.play().catch(() => {});
					azanPlayed = p.key;
				}
			}
		}, 1000);

		// ─── CALENDAR ───
		function showWeek() {
			$('tab-week').classList.add('active');
			$('tab-month').classList.remove('active');
			const cal = $('calendar');
			cal.innerHTML = '';
			const now = new Date(),
				start = new Date(now);
			start.setDate(now.getDate() - now.getDay());
			for (let i = 0; i < 14; i++) {
				const date = new Date(start);
				date.setDate(start.getDate() + i);
				const isToday = date.toDateString() === new Date().toDateString();
				const el = document.createElement('div');
				el.className = `cal-day${isToday?' active':''}`;
				el.innerHTML = `<span class="cal-wday">${date.toLocaleDateString('en-US',{weekday:'short'})}</span><span class="cal-num">${date.getDate()}</span>`;
				el.addEventListener('click', () => {
					document.querySelectorAll('.cal-day').forEach(e => e.classList.remove('active'));
					el.classList.add('active');
					fetchPrayers(date);
				});
				cal.appendChild(el);
			}
		}

		function showMonth() {
			$('tab-week').classList.remove('active');
			$('tab-month').classList.add('active');
			const cal = $('calendar');
			cal.innerHTML = '';
			const now = new Date(),
				yr = now.getFullYear(),
				mo = now.getMonth();
			const days = new Date(yr, mo + 1, 0).getDate(),
				first = new Date(yr, mo, 1).getDay();
			for (let i = 0; i < first; i++) {
				const e = document.createElement('div');
				e.className = 'cal-day';
				e.style.visibility = 'hidden';
				cal.appendChild(e);
			}
			for (let d = 1; d <= days; d++) {
				const date = new Date(yr, mo, d);
				const el = document.createElement('div');
				el.className = `cal-day${d===now.getDate()?' active':''}`;
				el.innerHTML = `<span class="cal-wday">${date.toLocaleDateString('en-US',{weekday:'short'})}</span><span class="cal-num">${d}</span>`;
				el.addEventListener('click', () => {
					document.querySelectorAll('.cal-day').forEach(e => e.classList.remove('active'));
					el.classList.add('active');
					fetchPrayers(date);
				});
				cal.appendChild(el);
			}
		}

				azanAudio.currentTime = 0;

		}, {
			once: true
		});
		setInterval(() => {

		setInterval(() => {
			if (userCoords) fetchWeather(userCoords.latitude, userCoords.longitude);
		}, 600000);
		setInterval(() => {
			if (todayPrayers.length) renderGrid();

		(function() {
			const surahs = ["Al-Fatiha", "Al-Baqarah", "Al-Imran", "An-Nisa", "Al-Ma'idah", "Al-An'am", "Al-A'raf", "Al-Anfal", "At-Tawbah", "Yunus", "Hud", "Yusuf", "Ar-Ra'd", "Ibrahim", "Al-Hijr", "An-Nahl", "Al-Isra", "Al-Kahf", "Maryam", "Ta-Ha", "Al-Anbiya", "Al-Hajj", "Al-Mu'minun", "An-Nur", "Al-Furqan", "Ash-Shu'ara", "An-Naml", "Al-Qasas", "Al-Ankabut", "Ar-Rum", "Luqman", "As-Sajda", "Al-Ahzab", "Saba", "Fatir", "Ya-Sin", "As-Saffat", "Sad", "Az-Zumar", "Ghafir", "Fussilat", "Ash-Shura", "Az-Zukhruf", "Ad-Dukhan", "Al-Jathiya", "Al-Ahqaf", "Muhammad", "Al-Fath", "Al-Hujurat", "Qaf", "Adh-Dhariyat", "At-Tur", "An-Najm", "Al-Qamar", "Ar-Rahman", "Al-Waqia", "Al-Hadid", "Al-Mujadila", "Al-Hashr", "Al-Mumtahina", "As-Saff", "Al-Jumu'a", "Al-Munafiqun", "At-Taghabun", "At-Talaq", "At-Tahrim", "Al-Mulk", "Al-Qalam", "Al-Haqqah", "Al-Ma'arij", "Nuh", "Al-Jinn", "Al-Muzzammil", "Al-Muddaththir", "Al-Qiyamah", "Al-Insan", "Al-Mursalat", "An-Naba", "An-Nazi'at", "Abasa", "At-Takwir", "Al-Infitar", "Al-Mutaffifin", "Al-Inshiqaq", "Al-Buruj", "At-Tariq", "Al-A'la", "Al-Ghashiya", "Al-Fajr", "Al-Balad", "Ash-Shams", "Al-Lail", "Ad-Duha", "Ash-Sharh", "At-Tin", "Al-Alaq", "Al-Qadr", "Al-Bayyina", "Az-Zalzalah", "Al-Adiyat", "Al-Qari'a", "At-Takathur", "Al-Asr", "Al-Humaza", "Al-Fil", "Quraish", "Al-Ma'un", "Al-Kawthar", "Al-Kafirun", "An-Nasr", "Al-Masad", "Al-Ikhlas", "Al-Falaq", "An-Nas"];
			const TOTAL = surahs.length;

				unlocked = false;

				idx = i;
				$('surah-name').textContent = surahs[i - 1];
				$('p-status').textContent = 'Loading…';
				$('progress').value = 0;
				qa.src = `quran/${String(i).padStart(3,'0')}.mp3`;
				qa.load();
				localStorage.setItem('surahIndex', i);
			}
			const doPlay = () => qa.play().then(() => {
				$('play').textContent = '⏸';
				playing = true;
				$('p-status').textContent = 'Playing';
			}).catch(() => $('p-status').textContent = 'Tap ▶');
			const doPause = () => {
				qa.pause();
				$('play').textContent = '▶';
				playing = false;
				$('p-status').textContent = 'Paused';

				if (unlocked && playing) setTimeout(doPlay, 500);

			$('prev').onclick = () => {
				load(idx === 1 ? TOTAL : idx - 1);
				if (unlocked && playing) setTimeout(doPlay, 500);
			};
			$('vol-range').oninput = () => qa.volume = $('vol-range').value;
			qa.ontimeupdate = () => {
				if (qa.duration) $('progress').value = (qa.currentTime / qa.duration) * 100;

			$('progress').oninput = () => {
				if (qa.duration) qa.currentTime = ($('progress').value / 100) * qa.duration;
			};
			qa.onloadedmetadata = () => $('p-status').textContent = 'Ready';
			qa.onended = () => $('next').click();
			qa.onerror = () => $('p-status').textContent = '❌ Not found';
			load(idx);
		})();
